<nlog xmlns="http://www.nlog-project.org/schemas/NLog.xsd"
      xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance">

	<targets>
		<!-- Existing Database Target -->
		<target name="database" xsi:type="Database">
			<commandText>
				insert into dbo.ErrorLog (
				ExceptionType, ExceptionMessage, ExceptionSource, ExceptionStacktrace,
				Host, IP, Browser, UserID, UserType, PageName, ErrorDate
				)
				values (
				@ExceptionType, @ExceptionMessage, @ExceptionSource, @ExceptionStacktrace,
				@Host, @IP, @Browser, @UserID, @UserType, @PageName, GETDATE()
				);
			</commandText>

			<parameter name="@ExceptionType" layout="${exception:format=Type}" />
			<parameter name="@ExceptionMessage" layout="${exception:format=Message}" />
			<parameter name="@ExceptionSource" layout="${exception:format=Source}" />
			<parameter name="@ExceptionStacktrace" layout="${exception:format=StackTrace}" />
			<parameter name="@Host" layout="${machinename}" />
			<parameter name="@PageName" layout="${event-properties:item=PageName}" />
			<parameter name="@IP" layout="${event-properties:item=IP}" />
			<parameter name="@Browser" layout="${event-properties:item=Browser}" />
			<parameter name="@UserID" layout="${event-properties:item=UserID}" />
			<parameter name="@UserType" layout="${event-properties:item=UserType}" />

			<dbProvider>System.Data.SqlClient</dbProvider>
		</target>

		<!-- File Target - Simple Format -->
		<target name="logfile" xsi:type="File"
		        fileName="${basedir}/Logs/${shortdate}.log"
		        layout="${longdate}|${level:uppercase=true}|${logger}|${message}|${exception:format=tostring}" />

		<!-- File Target - Detailed Format (Recommended) -->
		<target name="detailedfile" xsi:type="File"
		        fileName="${basedir}/Logs/${shortdate}_detailed.log"
		        archiveFileName="${basedir}/Logs/Archive/${shortdate}.{#}.log"
		        archiveEvery="Day"
		        archiveNumbering="Rolling"
		        maxArchiveFiles="30"
		        concurrentWrites="true"
		        keepFileOpen="false">
			<layout xsi:type="JsonLayout">
				<attribute name="time" layout="${longdate}" />
				<attribute name="level" layout="${level:upperCase=true}"/>
				<attribute name="message" layout="${message}" />
				<attribute name="exception" layout="${exception:format=tostring}" />
				<attribute name="exceptionType" layout="${exception:format=Type}" />
				<attribute name="exceptionMessage" layout="${exception:format=Message}" />
				<attribute name="stackTrace" layout="${exception:format=StackTrace}" />
				<attribute name="host" layout="${machinename}" />
				<attribute name="pageName" layout="${event-properties:item=PageName}" />
				<attribute name="ip" layout="${event-properties:item=IP}" />
				<attribute name="browser" layout="${event-properties:item=Browser}" />
				<attribute name="userId" layout="${event-properties:item=UserID}" />
				<attribute name="userType" layout="${event-properties:item=UserType}" />
				<attribute name="logger" layout="${logger}"/>
			</layout>
		</target>

		<!-- File Target - Error Only (Separate File) -->
		<target name="errorfile" xsi:type="File"
		        fileName="${basedir}/Logs/Errors/${shortdate}_errors.log"
		        layout="${longdate} | ${level:uppercase=true} | ${machinename} | ${message} ${newline}Exception: ${exception:format=tostring}${newline}PageName: ${event-properties:item=PageName}${newline}UserID: ${event-properties:item=UserID}${newline}IP: ${event-properties:item=IP}${newline}Browser: ${event-properties:item=Browser}${newline}---${newline}" />

	</targets>

	<rules>
		<!-- Log errors to database -->
		<logger name="*" minlevel="Error" writeTo="database" />
		
		<!-- Log errors to file -->
		<logger name="*" minlevel="Error" writeTo="errorfile" />
		
		<!-- Log all levels to detailed file (optional) -->
		<logger name="*" minlevel="Info" writeTo="detailedfile" />
		
		<!-- Or just log everything to simple file -->
		<!-- <logger name="*" minlevel="Trace" writeTo="logfile" /> -->
	</rules>
</nlog>
