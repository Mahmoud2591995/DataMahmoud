<input
  type="text"
  name="numberField"
  [(ngModel)]="requestModel.NumberField"
  #numberField="ngModel"
  pattern="[0-9]{5,15}"
  minlength="5"
  maxlength="15"
  required
  class="form-control"
  (input)="onNumberInput($event)"
  [disabled]="currentPageMode === 3"
/>

<small class="text-danger" *ngIf="numberField.errors?.['minlength'] && numberField.touched">
  Minimum 5 digits required
</small>
<small class="text-danger" *ngIf="numberField.errors?.['maxlength'] && numberField.touched">
  Maximum 15 digits allowed
</small>
<small class="text-danger" *ngIf="numberField.errors?.['pattern'] && numberField.touched">
  Only numbers allowed
</small>

onNumberInput(event: Event) {
  const input = event.target as HTMLInputElement;
  input.value = input.value.replace(/[^0-9]/g, '');
}
